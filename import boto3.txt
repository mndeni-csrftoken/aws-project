import boto3
import os
from PIL import Image
from io import BytesIO
import uuid
from datetime import datetime

s3 = boto3.client('s3')
dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('ImageMetadata')

PROCESSED_BUCKET = 'processed-images-<your-name>'

def lambda_handler(event, context):
    # 1. Get S3 event info
    record = event['Records'][0]
    source_bucket = record['s3']['bucket']['name']
    source_key = record['s3']['object']['key']

    # 2. Download image from S3
    obj = s3.get_object(Bucket=source_bucket, Key=source_key)
    image_content = obj['Body'].read()

    # 3. Open and resize image
    image = Image.open(BytesIO(image_content))
    image.thumbnail((200, 200))  # 200x200 thumbnail

    # 4. Save thumbnail to memory
    buffer = BytesIO()
    image.save(buffer, format='JPEG')
    buffer.seek(0)

    # 5. Generate new key for processed image
    processed_key = f"thumbnails/{os.path.basename(source_key)}"

    # 6. Upload thumbnail to processed bucket
    s3.put_object(
        Bucket=PROCESSED_BUCKET,
        Key=processed_key,
        Body=buffer,
        ContentType='image/jpeg'
    )

    # 7. Store metadata in DynamoDB
    image_id = str(uuid.uuid4())
    table.put_item(
        Item={
            'imageId': image_id,
            'originalBucket': source_bucket,
            'originalKey': source_key,
            'processedBucket': PROCESSED_BUCKET,
            'processedKey': processed_key,
            'createdAt': datetime.utcnow().isoformat()
        }
    )

    return {
        'statusCode': 200,
        'body': f"Processed image stored at {PROCESSED_BUCKET}/{processed_key}"
    }
